## 3. 数据模型关系图

```mermaid
erDiagram
    USER ||--o{ PROJECT_MEMBER : "参与"
    PROJECT ||--o{ PROJECT_MEMBER : "拥有"
    PROJECT ||--o{ REQUIREMENT : "包含"
    PROJECT ||--o{ ARCHITECTURE : "拥有"
    PROJECT ||--o{ DOCUMENT : "生成"
    
    REQUIREMENT ||--o{ USER_STORY : "关联用例"
    REQUIREMENT ||--o{ QUALITY_EVALUATION : "质量评估"
    REQUIREMENT ||--o{ USE_CASE : "用例"
    
    ARCHITECTURE ||--o{ ADR : "架构决策"
    ARCHITECTURE ||--o{ TRACEABILITY : "追溯映射"
    
    DOCUMENT ||--o{ OPERATION_LOG : "操作日志"
    DOCUMENT ||--o{ ATTACHMENT : "附件"

    USER {
        uuid id PK
        string username UK
        string email UK
        string hashed_password
        string full_name
        enum role "RE/SA/ADMIN"
        boolean is_active
        timestamp created_at
    }

    PROJECT {
        uuid id PK
        string name
        text description
        uuid owner_id FK
        timestamp created_at
    }

    REQUIREMENT {
        uuid id PK
        uuid project_id FK
        string title
        text description
        enum type "FR/NFR"
        enum category
        enum status "DRAFT/CONFIRMED"
        boolean is_ai_extracted
        timestamp created_at
    }

    ARCHITECTURE {
        uuid id PK
        uuid project_id FK
        jsonb components
        jsonb patterns
        text rationale
        integer version
        timestamp created_at
    }

    DOCUMENT {
        uuid id PK
        uuid project_id FK
        string doc_type "SRS/SAD"
        text content
        integer version
        timestamp created_at
    }
```
